include(ExternalProject)
ExternalProject_Add(
  ipaddress

  GIT_REPOSITORY "https://github.com/mabels/ipaddress.git"
  GIT_TAG "master"


  SOURCE_DIR "${CMAKE_SOURCE_DIR}/3rdparty/ipaddress"
  CMAKE_ARGS -DCMAKE_BUILD_TYPE=debug -DBUILD_STATIC_LIBS=ON -DBUILD_SHARED_LIBS=OFF -DARCHIVE_INSTALL_DIR=<INSTALL_DIR> -G "Unix Makefiles" -DCMAKE_INSTALL_PREFIX:PATH=<INSTALL_DIR> ../..

  #UPDATE_COMMAND "${CMAKE_COMMAND} cpp"
  #BUILD_COMMAND "make"
  INSTALL_COMMAND ""
  TEST_COMMAND ""
)


ExternalProject_Get_Property(ipaddress install_dir)
include_directories(${install_dir}/include)

set(ipaddress_INCLUDE_DIRS "${CMAKE_SOURCE_DIR}/3rdparty/ipaddress/cpp/src")
set(ipaddress_LIBRARIES "${CMAKE_SOURCE_DIR}/3rdparty/ipaddress/cpp/${CMAKE_BUILD_TYPE}/${CMAKE_SYSTEM_NAME}/${CMAKE_STATIC_LIBRARY_PREFIX}ipaddress${CMAKE_STATIC_LIBRARY_SUFFIX}")
include_directories(${ipaddress_INCLUDE_DIRS})
