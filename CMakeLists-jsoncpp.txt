include(ExternalProject)
ExternalProject_Add(
  jsoncpp

  GIT_REPOSITORY "https://github.com/open-source-parsers/jsoncpp.git"
  GIT_TAG "master"

  UPDATE_COMMAND ""

  SOURCE_DIR "${CMAKE_SOURCE_DIR}/3rdparty/jsoncpp"
  CMAKE_ARGS -DCMAKE_BUILD_TYPE=debug -DBUILD_STATIC_LIBS=ON -DBUILD_SHARED_LIBS=OFF -DARCHIVE_INSTALL_DIR=. -G "Unix Makefiles" ../..


  BUILD_COMMAND ""
  INSTALL_COMMAND ""
  TEST_COMMAND ""
)



set(jsoncpp_INCLUDE_DIRS "${CMAKE_SOURCE_DIR}/3rdparty/jsoncpp/include")
include_directories(${jsoncpp_INCLUDE_DIRS})
